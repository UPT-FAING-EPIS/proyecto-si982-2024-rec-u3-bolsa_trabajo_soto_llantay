

CREATE DATABASE BD_BOLSA_TRABAJO

GO

USE BD_BOLSA_TRABAJO

CREATE TABLE USUARIOS (
    COD_USUARIO INT PRIMARY KEY IDENTITY(1,1),
    USUARIO VARCHAR(50) NOT NULL,
    CONTRASEÑA VARCHAR(255) NOT NULL,
    CARGO VARCHAR(50) NOT NULL
);

CREATE TABLE LUGAR (
    ID_LUGAR INT PRIMARY KEY IDENTITY(1,1),
    NOMBRE VARCHAR(50) NOT NULL
);

CREATE TABLE AREA (
    ID_AREA INT PRIMARY KEY IDENTITY(1,1),
    NOMBRE VARCHAR(50) NOT NULL
);

CREATE TABLE TRABAJO (
    ID_TRABAJO INT PRIMARY KEY IDENTITY(1,1),
    FECHA_PUBLICACION  varchar(50) NOT NULL,
    ID_AREA INT NOT NULL,
    ID_LUGAR INT NOT NULL,
    EMPRESA VARCHAR(70) NOT NULL,
    TITULO VARCHAR(100) NOT NULL,
    RESUMEN varchar(400) NOT NULL,
    REQUERIMIENTO  varchar(200) NOT NULL,
    OFERTA DECIMAL(12,2) NOT NULL,
    COD_USUARIO INT NOT NULL,
    FOREIGN KEY (ID_AREA) REFERENCES AREA(ID_AREA),
    FOREIGN KEY (ID_LUGAR) REFERENCES LUGAR(ID_LUGAR),
    FOREIGN KEY (COD_USUARIO) REFERENCES USUARIOS(COD_USUARIO)
);

CREATE TABLE POSTULANTE (
    ID_POSTULANTE INT PRIMARY KEY IDENTITY(1,1),
    FECHA_POSTULACION varchar(50) NOT NULL,
    ID_TRABAJO INT NOT NULL,
    ARCHIVO nvarchar(150) NOT NULL,
    FOREIGN KEY (ID_TRABAJO) REFERENCES TRABAJO(ID_TRABAJO)
);
